Table seasons {
  season integer [primary key]
}

Table races {
  race_id integer [primary key]
  season integer
  round integer
  race_name varchar
  circuit_id integer
  race_date date
}

Table circuits {
  circuit_id integer [primary key]
  circuit_ref varchar
  circuit_name varchar
  circuit_country varchar
}

Table drivers {
  driver_id integer [primary key]
  driver_ref varchar
  first_name varchar
  last_name varchar
  driver_nationality varchar
  dob date
}

Table constructors {
  constructor_id integer [primary key]
  constructor_ref varchar
  constructor_name varchar
  constructor_nationality varchar
}

Table quali {
  quali_id integer [primary key]
  race_id integer
  driver_id integer
  constructor_id integer
  grid_position integer
  q1 integer
  q2 integer
  q3 integer
}

Table results {
  result_id integer [primary key]
  race_id integer
  driver_id integer
  constructor_id integer
  grid_position integer
  final_position integer
  points float
  time varchar
}

Table sprint_results {
  sprint_result_id integer [primary key]
  race_id integer
  driver_id integer
  constructor_id integer
  grid_position integer
  final_position integer
  points float
  time varchar
}

Ref: seasons.season > races.season
Ref: circuits.circuit_id > races.circuit_id

Ref: races.race_id > quali.race_id
Ref: drivers.driver_id > quali.driver_id
Ref: constructors.constructor_id > quali.constructor_id

Ref: races.race_id > results.race_id
Ref: drivers.driver_id > results.driver_id
Ref: constructors.constructor_id > results.constructor_id

Ref: races.race_id > sprint_results.race_id
Ref: drivers.driver_id > sprint_results.driver_id
Ref: constructors.constructor_id > sprint_results.constructor_id
